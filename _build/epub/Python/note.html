#初始化三维维数组2*3*4
p3=[[[1] * 2 for i in xrange(3)] for j in xrange(4)]
#Or
p3=[[[1 for i in xrange(2)] for j in xrange(3)] for k in xrange(4)]
#Or
p3=[]
for i in range(2):
    p3.append([])
    for j in range(3):
        p3[i].append([])
        for k in range(4):
            p3[i][j].append(0)
